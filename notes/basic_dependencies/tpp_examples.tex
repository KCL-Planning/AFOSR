\subsection*{Properties}

\begin{itemize}
	\item buy goods x in market y ($b(G_x, M_y)$)
	\item load goods x in market y ($l(G_x, M_y)$)
	\item buy goods x before monylevel is below y ($bb(G_x, ML_y)$)
	\item if you buy goods a in market b then buy goods c in market d ($bd(G_a, M_b, G_c, M_d)$)
	\item buy goods x and goods y in the same market ($sm(G_x, G_y)$)
\end{itemize}
